经典排序算法

插入排序

//不改变输入数组的状态，返回排序完成的降序数组
func insertionSort(var arr:[Int]) -> [Int] {
    for var i = 1; i < arr.count; i++ {
        let key = arr[i]
        var j = i - 1
        while j >= 0  && arr[j] > key {
            arr[j+1] = arr[j]
            j--
        }
        arr[j+1] = key
    }
    return arr
}
